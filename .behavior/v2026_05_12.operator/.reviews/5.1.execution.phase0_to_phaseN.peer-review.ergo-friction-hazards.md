The query.database skill (user-faced CLI operation) has no --help handling. The arg parser only looks for --sql/--format and skips other flags without validation or usage output. Passing --help or invalid args falls through to the 'absent required arg: --sql' error. Per the rule, --help must exist and show all options; required vs optional inputs must be clear. This creates flow friction and violates help/discoverability checks.

**snippet**:
```ts

    const arg = args[i]!;
    if (arg === '--sql') {
      const value = args[++i];
      if (value === '@stdin') {
        // read from stdin
        const chunks: Buffer[] = [];
        for await (const chunk of process.stdin) {
          chunks.push(chunk);
        }
        sql = Buffer.concat(chunks).toString('utf-8').trim();
      } else {
        sql = value ?? '';
      }
    } else if (arg === '--format') {
      format = args[++i] ?? 'table';
    } else if (
      ['--env', '--limit', '--repo', '--role', '--skill'].includes(arg)
    ) {
      i++; // skip value
    } else if (arg === '--') {
      // skip separator
    }
  }
```
