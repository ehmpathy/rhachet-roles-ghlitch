given(getS3BucketExpenseEvaluator emits json list of buckets that are expensive)
  when([t1] we invoke setS3BucketTierOptimization with --mode=prep)
    then(it grabs buckets that cost more than $5/mo in storage OR >5% of total S3 spend)
    then(it checks for existing lifecycle policies and flags conflicts)
    then(it counts total objects in the bucket)
    then(it evaluates, via list-objects, how many objects are >= 128KB in size)
    then(it calculates average object size)
    then(it estimates monitoring fees = (object_count / 1000) * 0.0025 per month)
    then(it emits a report of what % of objects are >= 128KB)
    then(it emits a report of the top 1k most common directory prefixes and their storage size; treestruct style)
    then(it emits a report of a random sample of most frequently accessed objects)
    then(it emits access pattern report: % accessed in 3d, 30d, 90d, 180d, 365d when available)
    then(it calculates current monthly cost from expense evaluator data)
    then(it estimates IT costs based on access patterns)
    then(it calculates expected monthly savings = current_cost - estimated_it_cost)
    then(it calculates expected savings percentage)
    then(it determines recommendation strategy based on:
      - conservative: if avg_size > 128KB AND >30% unaccessed 90d AND savings > 20%
      - aggressive: if avg_size > 128KB AND >60% unaccessed 90d AND bucket type=logs|archive
      - none: if avg_size < 128KB OR monitoring_fees > savings OR >70% accessed 30d)
    then(all reports emitted as markdown files to .rhachet/s3-tier-optimization/{bucket}/prep.md)
    then(it creates summary CSV with all buckets and their recommendations)

  when([t2] we invoke setS3BucketTierOptimization with --mode=exec)
    then(it loads the prep report for specified bucket)
    then(it checks that recommendation exists and is not "none")
    then(if recommendation is "none", it fails fast with explanation)
    then(if bucket is production (tag:env=prod), it requires --confirm flag)
    then(it backs up existing lifecycle policy to .rhachet/s3-tier-optimization/{bucket}/backup.json)
    then(it applies the lifecycle policy to transition to INTELLIGENT_TIERING)
    then(it applies the appropriate archive tier configuration per strategy)
    then(it emits confirmation with expected tier transitions)
    then(it logs action to .rhachet/s3-tier-optimization/applied.log)
