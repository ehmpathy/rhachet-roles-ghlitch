given(an aws account with lambda functions)
  when([t1] asked to set lambda observability settings in --mode=prep)
    then(enumerates all lambda functions)
    then(identifies which functions lack lambda insights)
    then(shares current observability settings per function)
    then(declares the diff after enabling insights on all functions)
  when([t2] asked to set lambda observability settings in --mode=exec)
    then(enumerates all lambda functions)
    then(enables lambda insights on functions that lack it)
    then(verifies the configuration was applied successfully)
    then(generates report of updated functions)
