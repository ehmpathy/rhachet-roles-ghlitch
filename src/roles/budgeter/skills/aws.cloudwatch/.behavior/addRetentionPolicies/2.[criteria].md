given(an aws account with cloudwatch log groups)
  when([t1] asked to add retention policies in --mode=prep)
    then(enumerates all of the cloudwatch log groups)
    then(shares the current policy of each)
    then(declares the diff after addition of policy to keep only for 90days)
  when([t2] asked to add retention policies in --mode=exec)
    then(enumerates all of the cloudwatch log groups)
    then(applies the policy to each of them)
