# diagnose.queryLogs

## acceptance.criteria::Behavior

usecase.1('propose which log groups to search')
- given('a list of log groups in an env.server=lambda project')
  - when('asked which log groups to query related to an error with @requestId=xyz')
    - then('it should choose all of them')
  - when('asked which log groups to query to detect what crontasks ping the database and at what cadence')
    - then('it should choose all of them')


usecase.2('propose what query to search with')
- given('a log pattern emitted whenever the database is queried')
  - when('asked to detect crontasks that hit the database')
    - then('it should query against that pattern')
    - then('it should order endpoints by frequency')
    - then('it should show average rate, for added context')
