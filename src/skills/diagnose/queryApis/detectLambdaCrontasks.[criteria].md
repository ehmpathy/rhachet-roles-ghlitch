usecase.1(detect lambda crontasks)
- given('exists lambdas, with and without crontasks')
  - when('plan: asked to identify lambdas with crontasks')
    - then('proposes aws cli command to enumerate all lambdas')
    - then('proposes aws cli command to check triggers for each lambda')
    - then('proposes jq filter to the ones with crontask triggers')
  - when('exec: executed the plan')
    - then('does not have any false positives; no lambdas that dont truely have crontasks')
    - then('does not have any false negatives; no lambdas that were missed with crontasks')
